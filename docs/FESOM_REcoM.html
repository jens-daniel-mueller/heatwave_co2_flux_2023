  # Climatoligcal mean
  
  df_mean <-
    df %>%
    filter(year < 2023,
           year >= 2023-5) %>%
    group_by(name, !!!group_by) %>%
    summarize(
      fit = mean(value, na.rm = TRUE)
    ) %>% 
    ungroup()
  
  
  df_mean <-
    full_join(
      df_mean,
      df
    ) %>%
    mutate(resid = value - fit)
  
  return(df_mean)
